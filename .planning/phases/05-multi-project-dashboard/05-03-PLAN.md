```tsx
function MetadataField({
  label,
  value,
  onSave,
}: {
  label: string;
  value: string | null;
  onSave: (newValue: string | null) => void;
}) {
  const [editing, setEditing] = useState(false);
  const [draft, setDraft] = useState(value ?? '');
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    if (editing) {
      inputRef.current?.focus();
      inputRef.current?.select();
    }
  }, [editing]);

  const handleSave = () => {
    const trimmed = draft.trim();
    onSave(trimmed || null); // empty -> null for "needs info"
    setEditing(false);
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      e.preventDefault();
      handleSave();
    } else if (e.key === 'Escape') {
      setDraft(value ?? '');
      setEditing(false);
    }
  };

  const startEditing = () => {
    setDraft(value ?? '');
    setEditing(true);
  };

  return (
    <div className="py-2">
      <label className="text-xs font-medium text-slate-400">{label}</label>
      {editing ? (
        <input
          ref={inputRef}
          value={draft}
          onChange={(e) => setDraft(e.target.value)}
          onBlur={handleSave}
          onKeyDown={handleKeyDown}
          className="mt-0.5 w-full rounded border border-indigo-300 px-2 py-1 text-sm text-slate-700 outline-none focus:ring-2 focus:ring-indigo-500"
          aria-label={label}
        />
      ) : (
        <button
          onClick={startEditing}
          className="mt-0.5 block w-full text-left text-sm"
          title={`Edit ${label.toLowerCase()}`}
        >
          {value ? (
            <span className="text-slate-700">{value}</span>
          ) : (
            <span className="italic text-amber-500">Needs info</span>
          )}
        </button>
      )}
    </div>
  );
}
```

```tsx
import { useState, useRef, useEffect } from 'react';
import { useProjects } from '@/contexts/ProjectsContext';
import type { ProjectMetadata } from '@/services/storageService.types';

interface MetadataPanelProps {
  projectId: string;
}

export default function MetadataPanel({ projectId }: MetadataPanelProps) {
  const { state, updateProject } = useProjects();
  const project = state.projects.find((p) => p.id === projectId);

  if (!project) return null;

  const handleFieldSave = (
    field: keyof ProjectMetadata,
    value: string | null
  ) => {
    updateProject({ ...project, [field]: value });
  };

  // Format file size
  const formatSize = (bytes: number): string => {
    if (bytes < 1024) return `${bytes} B`;
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
    return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
  };

  // Format duration
  const formatDuration = (seconds: number): string => {
    if (seconds <= 0) return 'Unknown';
    const hrs = Math.floor(seconds / 3600);
    const mins = Math.floor((seconds % 3600) / 60);
    const secs = Math.floor(seconds % 60);
    if (hrs > 0) return `${hrs}h ${mins}m`;
    if (mins > 0) return `${mins}m ${secs}s`;
    return `${secs}s`;
  };

  // Status badge styles
  const statusStyles: Record<string, string> = {
    completed: 'bg-green-100 text-green-700',
    error: 'bg-red-100 text-red-700',
    idle: 'bg-slate-100 text-slate-600',
    uploading: 'bg-blue-100 text-blue-700',
    processing: 'bg-amber-100 text-amber-700',
  };

  return (
    <aside className="w-72 overflow-y-auto border-l border-slate-200 bg-white">
      <div className="p-4">
        {/* Header */}
        <h3 className="mb-4 text-xs font-semibold uppercase tracking-wider text-slate-400">
          Details
        </h3>

            label="Interviewee"
            value={project.interviewee ?? null}
            onSave={(v) => handleFieldSave('interviewee', v)}
          />
          <MetadataField
            label="Interviewer"
            value={project.interviewer ?? null}
            onSave={(v) => handleFieldSave('interviewer', v)}
          />
          <MetadataField
            label="Participants"
            value={project.participants ?? null}
            onSave={(v) => handleFieldSave('participants', v)}
          />
          <MetadataField
            label="Interview Date"
            value={project.interviewDate ?? null}
            onSave={(v) => handleFieldSave('interviewDate', v)}
          />
          <MetadataField
            label="Original Language"
            value={project.originalLanguage ?? null}
            onSave={(v) => handleFieldSave('originalLanguage', v)}
          />

            label="Location"

              >
                {project.status}
              </span>
            </p>
          </div>
```
